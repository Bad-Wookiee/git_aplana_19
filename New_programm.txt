add 'old directory'
add 'new directory'
add 'remaining time' = 'Осталось __ минут'
add 'error' = 'Ошибка загрузки, повторите попытку' 


\\open: folder 'old directory';

\\select: tag <new files> as 'fresh'
        \\move: 'fresh' to 'new directory';

\\show: 'remaining time';

\\if done
        \\open: folder 'new directory';

\\else
        \\show: 'error';
